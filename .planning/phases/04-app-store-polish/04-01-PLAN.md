For all three PNGs, use this Python script approach to create valid PNG files:

```python
import struct, zlib

def create_png(width, height, r, g, b, filename):
    def make_png(w, h, r, g, b):
        def png_chunk(chunk_type, data):
            c = chunk_type + data
            return struct.pack('>I', len(data)) + c + struct.pack('>I', zlib.crc32(c) & 0xffffffff)

        sig = b'\x89PNG\r\n\x1a\n'
        ihdr_data = struct.pack('>IIBBBBB', w, h, 8, 2, 0, 0, 0)
        ihdr = png_chunk(b'IHDR', ihdr_data)

        # Build image data: each row is filter_byte(0) + RGB pixels
        raw = b''
        row = bytes([r, g, b] * w)
        for _ in range(h):
            raw += b'\x00' + row

        idat = png_chunk(b'IDAT', zlib.compress(raw))
        iend = png_chunk(b'IEND', b'')
        return sig + ihdr + idat + iend

    with open(filename, 'wb') as f:
        f.write(make_png(width, height, r, g, b))

# Blue: #1a73e8 = RGB(26, 115, 232)
import os

create_png(1000, 700, 26, 115, 232, 'assets/images/xlarge.png')
print('PNG files created')
```
